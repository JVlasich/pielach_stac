\begin{enumerate}[label=\bfseries\arabic*.]
    \setcounter{enumi}{-1} % Startet die Nummerierung bei 0
    
    \item \textbf{Präambel: Zeitserie Pielach}
    \item \textbf{Einleitung}
    \item \textbf{Grundlagen und Stand der Technik}
    \begin{enumerate}[label=\arabic{enumi}.\arabic*]
        \item \textit{Cloud-Native-Geospatial} (COG, COPC und HTTP-Range-Requests)
        \item Metadatenstandards: Von ISO~19115 zu STAC
        \item Abgrenzung zum \textit{Data-Cube}-Paradigma
        \item Verwandte Arbeiten und Werkzeuge
    \end{enumerate}
    \item \textbf{Technische Ausgangslage der Rohdaten}
    \begin{enumerate}[label=\arabic{enumi}.\arabic*]
        \item Analyse der Verzeichnisstruktur und Dateiformate
        \item Metadaten-Inkonsistenzen und programmatische Herausforderungen
    \end{enumerate}
    \item \textbf{Konzeption der STAC-Architektur}
    \begin{enumerate}[label=\arabic{enumi}.\arabic*]
        \item Strukturierung der Katalog-Hierarchie
        \item Evaluierung der \textit{Linking}-Strategie
        \item Erweiterbarkeit und Idempotenz der Pipeline
        \item Architektur der Abfrage- und \textit{Subsetting}-Komponente
    \end{enumerate}
    \item \textbf{Implementierung}
    \begin{enumerate}[label=\arabic{enumi}.\arabic*]
        \item Datenaufbereitung und Überführung in \textit{Cloud-Native}-Formate
        \item Generierungspipeline (Metadatenextraktion, konfigurierbare \textit{Asset-Registry})
        \item Validierung (STAC-Konformität und inhaltliche Plausibilität)
        \item Domänenspezifische Abfragen und \textit{Subsetting}
    \end{enumerate}
    \item \textbf{Ergebnisse und Evaluierung}
    \begin{enumerate}[label=\arabic{enumi}.\arabic*]
        \item Präsentation des generierten STAC-Katalogs
        \item Domänenspezifische raumzeitliche Abfragen und \textit{Subsetting}
        \item Bewertung der Generalisierbarkeit der Pipeline
    \end{enumerate}
    \item \textbf{Diskussion}
    \begin{enumerate}[label=\arabic{enumi}.\arabic*]
        \item Architekturentscheidungen: Begründung und Übertragbarkeit
        \item Grenzen der Generalisierbarkeit und Konfigurierbarkeit
        \item Mehrwert gegenüber verzeichnisorientierter Ablage und FAIR-Bezug
    \end{enumerate}
    \item \textbf{Fazit und Ausblick}
\end{enumerate}
